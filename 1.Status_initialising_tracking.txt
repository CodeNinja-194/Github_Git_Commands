//status,Initializing,Tracking and Deleting 

1.git status   //first check if there are any files for any untracked files.

2.git init     //Initialized empty Git repository in particular location 

3.git add file_name //for commiting files or staging individual file.

4.git add --a  //for files into staging area

5.git add .  //all files into staging area

6.git commit -m "Initial commit" // message,commit all staged files to versioned history 

7.git commit  // open an editor 

8.git log //show commits,full change history

9.rm -rf .git //delete .git folder//stop tracking no longer a git repository.
